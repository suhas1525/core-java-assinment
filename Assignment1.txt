//3. Program to check whether number is prime or not.



class Assignment2

{

    public static void main(String args[])

    {

        Scanner sc=new Scanner(System.in);

        System.out.print("Enter a number: ");

        int no=sc.nextInt();

        sc.close();

        

        int i,sum=0;

        for(i=2;i<no;i++)

        {

            if(no%i==0)

            sum++;

        }

        if(sum>0)

               System.out.println(no+" is a Non-Prime number.");

        else

               System.out.println(no+" is a Prime number.");

    }

}

//5. Print all prime numbers between two given numbers. [break, continue]



class Assignment2

{

    public static void main(String args[])

    {

        Scanner sc=new Scanner(System.in);

        System.out.print("Enter Upper Bound: ");

        int lb=sc.nextInt();

        System.out.print("Enter Lower Bound: ");

        int ub=sc.nextInt();

        sc.close();

        

        int i,j,sum;

        System.out.print("Prime No. between "+lb+" and "+ub+" are :");



        for(i=lb+1;i<ub;i++)

        {

            sum=0;

            for(j=2;j<i;j++)

            {

                if(i%j==0)

                {

                    sum++;

                    break;

                }

                else

                    continue;

            }

            if(sum==0)

            System.out.print(" "+i+",");

        }

    }

}

//7. Sort a ten element array in descending order.



class Assignment2

{

    public static void main(String args[])

    {

        Scanner sc=new Scanner(System.in);

        int arr[]= new int[10];

        

        int i,j,y;

        for(i=0;i<5;i++)

        {

                System.out.print("Enter element "+(i+1)+": ");

                arr[i]=sc.nextInt();

        }

        sc.close();

        

        System.out.println("Decending order of Array: ");

        for(i=0;i<5;i++)

        {

                y=0;

                for(j=0;j<5;j++)

                {

                        if(arr[i]>arr[j] && arr[i]<y)

                        {

                            arr[i]=arr[j];

                            y=arr[j];

                        }

                }

        }

        

        for(i=4;i>=0;i--)

        {

                System.out.print(arr[i]);

        }

        System.out.print("\n");

    }

}

//9. Write a program to search an element in the array.



class Assignment2

{

    public static void main(String args[])

    {

        Scanner sc=new Scanner(System.in);

        int arr[]= new int[10];

        

        System.out.print("Enter element to be searched: ");

        int el=sc.nextInt();

        

        int i,sum=0,pos=0;

        for(i=0;i<10;i++)

        {

           System.out.print("Enter element "+(i+1)+": ");

           arr[i]=sc.nextInt();

        }

        sc.close();

        

        for(i=0;i<10;i++)

        {

            if(el==arr[i])

            {

                sum++;

                pos=(i+1);

                break;

            }

        }

        if(sum>0)

        System.out.println(el+" is present in the Array at position "+pos);

        else

        System.out.println(el+" is not present in the Array.");

    }

}
